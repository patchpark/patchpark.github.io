# 문제 링크
[1285](https://www.acmicpc.net/problem/1285)

# 접근 방법

뒤집을 행을 선택한 뒤(2^n) 뒤집고, 각 열에 대해 앞면과 뒷면 중 짧은 것을 골라 다 더해주면 된다. 뒤집을 행을 선택할 때 비트마스킹을 이용하는 다른 풀이들과 달리, dfs를 통해 선택해주었다.

# 문제 풀이
```java
import java.util.Scanner;

public class Main{
	
	static int n;
	static char[][] coin;
	static int min = Integer.MAX_VALUE;
	
	public static void main(String[] args) {
		
		Scanner sc = new Scanner(System.in);
		
		n = sc.nextInt();
		
		coin = new char[n][n];
		
		for(int i=0; i<n; i++) {
			String str = sc.next();
			for(int j=0; j<n; j++) {
				coin[i][j] = str.charAt(j);
			}
		}		
		rowToFlip(0);
		
		System.out.println(min);
	}
	
	public static void rowToFlip(int row) {
        if (row == n) {
            count();
            return;
        }
        rowToFlip(row + 1);
        
        for (int j = 0; j < n; j++) {
            coin[row][j] = flip(coin[row][j]);
        }
        rowToFlip(row + 1);
        
        for (int j = 0; j < n; j++) {
            coin[row][j] = flip(coin[row][j]);
        }
    }
	
	public static void count() {
		int total = 0;
		
		for(int i=0; i<n; i++) {
			int Tcount = 0;
			for(int j=0; j<n; j++) {
				if(coin[j][i] == 'T')
					Tcount++;
			}
			total += Math.min(Tcount, n - Tcount);	
		}
		if(total < min)
			min = total;
		
	}
	
	public static char flip(char c) {
		if(c == 'H')
			return 'T';
		else
			return 'H';
	}
}


```
